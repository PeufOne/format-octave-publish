Partie 1 : calcul simple

b = [4; 9; 2];

      1 3 1;
      3 5 9 ];

Partie 2 : sinus

figure(1)
x = -10:0.1:10;
y = sin (x);
plot (x, y);
title ("Simple 2-D Plot");
xlabel ("x");
ylabel ("sin (x)");
